```powershell
# Auto-sync a Google Drive cada hora
$watcher = New-Object System.IO.FileSystemWatcher
$watcher.Path = "C:\Users\Lenovo\.gemini\antigravity"
$watcher.IncludeSubdirectories = $true
$watcher.EnableRaisingEvents = $true

$action = {
    # Subir a Google Drive
    rclone copy $Event.FullPath "gdrive:TESIS_BACKUP/"
}
```
